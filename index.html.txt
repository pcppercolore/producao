    <div class="header">
        <h1 id="kanban-title">📋 Quadro de Produção</h1>
        <div>
            <button id="filtro-agitadores" class="filter-btn active" onclick="filtrar('Agitadores')">🌀 Agitadores</button>
            <button id="filtro-dosadoras" class="filter-btn" onclick="filtrar('Dosadoras')">🧪 Dosadoras</button>
        </div>
    </div>

    <div class="kanban-board" id="quadro"></div>

    <script>
        let filtroAtual = 'Agitadores';
        let todasMaquinas = [];

        const colunasAgitadores = ['Demanda da Semana', 'Em montagem', 'Teste', 'Adesivagem', 'Checklist', 'Embalagem'];
        const colunasDosadoras = ['Demanda da Semana', 'Em montagem', 'Check List', 'Adesivagem', 'Teste', 'Embalagem'];

        // Carrega o arquivo dados.json gerado pelo aplicativo admin
        async function buscarDados() {
            try {
                // O parâmetro '?t=' + Date.now() força o navegador a buscar o arquivo mais recente da branch principal
                const resposta = await fetch('dados.json?t=' + Date.now());
                const dados = await resposta.json();
                todasMaquinas = dados.maquinas || [];
                renderizarQuadro();
            } catch (erro) {
                console.error("Erro ao carregar dados do painel:", erro);
            }
        }

        function filtrar(tipo) {
            filtroAtual = tipo;
            document.getElementById('filtro-agitadores').classList.toggle('active', tipo === 'Agitadores');
            document.getElementById('filtro-dosadoras').classList.toggle('active', tipo === 'Dosadoras');
            renderizarQuadro();
        }

        function renderizarQuadro() {
            const quadro = document.getElementById('quadro');
            quadro.innerHTML = '';

            const colunas = (filtroAtual === 'Agitadores') ? colunasAgitadores : colunasDosadoras;
            document.getElementById('kanban-title').innerText = `📋 Produção: ${filtroAtual}`;

            // Criar as colunas industriais na tela
            colunas.forEach((nomeColuna, index) => {
                let cor = '#ffffff';
                if(nomeColuna === 'Demanda da Semana') cor = '#e67e22';
                if(nomeColuna === 'Em montagem') cor = '#3498db';
                if(nomeColuna === 'Check List' || nomeColuna === 'Checklist') cor = '#9b59b6';
                if(nomeColuna === 'Adesivagem') cor = '#00c0a5';
                if(nomeColuna === 'Teste') cor = '#ff4b55';
                if(nomeColuna === 'Embalagem') cor = '#2ecc71';

                quadro.innerHTML += `
                    <div class="kanban-column" style="border-top: 3px solid ${cor}">
                        <div class="column-title" style="color:${cor}">${nomeColuna}</div>
                        <div id="coluna-${index}"></div>
                    </div>
                `;
            });

            // Distribuir as máquinas nas colunas correspondentes
            const filtradas = todasMaquinas.filter(m => m.tipo === filtroAtual);
            filtradas.forEach(maquina => {
                let idx = colunas.indexOf(maquina.status);
                if (idx === -1) idx = 0;

                const cardHTML = `
                    <div class="kanban-card">
                        <div class="card-title">${maquina.modelo} (${maquina.condicao})</div>
                        <div class="card-info">
                            <strong>Série:</strong> ${maquina.serie}<br>
                            <strong>Cliente:</strong> ${maquina.cliente}<br>
                            <strong>Destino:</strong> ${maquina.cidade || 'Não informada'}-${maquina.estado || ''}<br>
                            <strong>Bandeira:</strong> ${maquina.bandeira}
                        </div>
                    </div>
                `;

                const colContainer = document.getElementById(`coluna-${idx}`);
                if (colContainer) colContainer.innerHTML += cardHTML;
            });
        }

        // Executa a busca assim que a página abre
        buscarDados();
        
        // Atualiza a tela automaticamente a cada 30 segundos
        setInterval(buscarDados, 30000);
    </script>
